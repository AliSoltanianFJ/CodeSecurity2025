@echo off
set REPO_DIR=Scenarios\JavaScript\o4_mini\Scenarios
set DB_NAME=my-codeql-db-js
set SARIF_OUT=results.sarif

echo === [1] Create CodeQL Database ===
codeql database create "%DB_NAME%" ^
  --language=javascript ^
  --source-root="%REPO_DIR%" ^
  --overwrite

echo === [2] Analyze with CodeQL ===
codeql database analyze my-codeql-db-js --format=sarif-latest --output=results.sarif --threads=4 "Scenarios\codeql-repo\javascript\ql\src\Security"

echo === Done! Results saved to %SARIF_OUT%
pause
exit /b
